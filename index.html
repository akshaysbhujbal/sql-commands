üî® DDL (Data Definition Language) Commands

CREATE - Create Database/Table
‚úì Fully Supported in MySQL
Creates a new database or table
-- Create Database
CREATE DATABASE school;

-- Create Table
CREATE TABLE students (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    age INT,
    email VARCHAR(100) UNIQUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
ALTER - Modify Table Structure
‚úì Fully Supported in MySQL 
-- Add Column
ALTER TABLE students ADD phone VARCHAR(15);
ALTER TABLE students ADD COLUMN address TEXT AFTER name;

-- Modify Column (change datatype)
ALTER TABLE students MODIFY COLUMN age TINYINT;
ALTER TABLE students MODIFY COLUMN name VARCHAR(150) NOT NULL;

-- Change Column (rename + modify)
ALTER TABLE students CHANGE COLUMN old_name new_name VARCHAR(100);
ALTER TABLE students CHANGE phone mobile VARCHAR(20);

-- Drop Column
ALTER TABLE students DROP COLUMN phone;

-- Rename Table
ALTER TABLE students RENAME TO learners;

-- Add Constraint
ALTER TABLE students ADD CONSTRAINT uk_email UNIQUE (email);
ALTER TABLE students ADD CONSTRAINT fk_dept FOREIGN KEY (dept_id) REFERENCES departments(id);

-- Drop Constraint
ALTER TABLE students DROP CONSTRAINT uk_email;
ALTER TABLE students DROP FOREIGN KEY fk_dept;

-- Add Primary Key
ALTER TABLE students ADD PRIMARY KEY (id);

-- Drop Primary Key
ALTER TABLE students DROP PRIMARY KEY;
DROP - Delete Database/Table
‚úì Fully Supported in MySQL 
-- Drop Table
DROP TABLE students;

-- Drop Database
DROP DATABASE school;

-- Drop if exists
DROP TABLE IF EXISTS students;
TRUNCATE - Remove All Records
‚úì Fully Supported in MySQL
Deletes all rows but keeps structure (faster than DELETE)
TRUNCATE TABLE students;
RENAME - Rename Table
‚úì Fully Supported in MySQL 
RENAME TABLE old_name TO new_name;

-- Multiple tables
RENAME TABLE students TO learners,
             courses TO subjects;

üìù DML (Data Manipulation Language) Commands

INSERT - Add New Records
‚úì Fully Supported in MySQL 
-- Insert single row
INSERT INTO students (name, age, email) 
VALUES ('John Doe', 20, 'john@email.com');

-- Insert multiple rows
INSERT INTO students (name, age, email) VALUES
('Jane Smith', 22, 'jane@email.com'),
('Bob Johnson', 21, 'bob@email.com');

-- Insert from another table
INSERT INTO students_backup 
SELECT * FROM students WHERE age > 18;
UPDATE - Modify Existing Records
‚úì Fully Supported in MySQL 
-- Update single record
UPDATE students 
SET age = 21 
WHERE id = 1;

-- Update multiple columns
UPDATE students 
SET age = 23, email = 'newemail@email.com' 
WHERE name = 'John Doe';

-- Update with calculation
UPDATE products 
SET price = price * 1.1;
DELETE - Remove Records
‚úì Fully Supported in MySQL 
-- Delete specific records
DELETE FROM students WHERE age < 18;

-- Delete all records (use TRUNCATE instead)
DELETE FROM students;

-- Delete with JOIN
DELETE students FROM students
JOIN enrollments ON students.id = enrollments.student_id
WHERE enrollments.status = 'inactive';
MERGE / UPSERT
‚úó MERGE Not Supported in MySQL
MySQL Alternative: Use INSERT ... ON DUPLICATE KEY UPDATE 
-- MySQL equivalent (upsert)
INSERT INTO students (id, name, age) 
VALUES (1, 'John', 21)
ON DUPLICATE KEY UPDATE 
    name = VALUES(name), 
    age = VALUES(age);

-- Or use REPLACE
REPLACE INTO students (id, name, age)
VALUES (1, 'John', 21);

üîç DQL (Data Query Language) Commands

SELECT - Retrieve Data
‚úì Fully Supported in MySQL 
-- Basic select
SELECT * FROM students;

-- Select specific columns
SELECT name, age FROM students;

-- With WHERE clause
SELECT * FROM students WHERE age > 20;

-- With ORDER BY
SELECT * FROM students ORDER BY age DESC;

-- With LIMIT
SELECT * FROM students LIMIT 10;

-- With OFFSET (pagination)
SELECT * FROM students LIMIT 10 OFFSET 20;

-- DISTINCT values
SELECT DISTINCT age FROM students;

-- Aggregate functions
SELECT COUNT(*), AVG(age), MAX(age), MIN(age), SUM(age)
FROM students;
WHERE Clause Operators
‚úì Fully Supported in MySQL 
-- Comparison operators
SELECT * FROM students WHERE age = 20;
SELECT * FROM students WHERE age != 20;
SELECT * FROM students WHERE age > 20;
SELECT * FROM students WHERE age >= 20;
SELECT * FROM students WHERE age < 20;
SELECT * FROM students WHERE age <= 20;

-- BETWEEN
SELECT * FROM students WHERE age BETWEEN 18 AND 25;

-- IN
SELECT * FROM students WHERE age IN (18, 20, 22);

-- LIKE (pattern matching)
SELECT * FROM students WHERE name LIKE 'J%';
SELECT * FROM students WHERE name LIKE '%son';
SELECT * FROM students WHERE name LIKE '%oh%';

-- IS NULL / IS NOT NULL
SELECT * FROM students WHERE email IS NULL;
SELECT * FROM students WHERE email IS NOT NULL;

-- AND, OR, NOT
SELECT * FROM students WHERE age > 20 AND name LIKE 'J%';
SELECT * FROM students WHERE age < 18 OR age > 25;
SELECT * FROM students WHERE NOT age = 20;
GROUP BY & HAVING
‚úì Fully Supported in MySQL 
-- Group by
SELECT age, COUNT(*) as count 
FROM students 
GROUP BY age;

-- Having (filter groups)
SELECT age, COUNT(*) as count 
FROM students 
GROUP BY age 
HAVING COUNT(*) > 5;

-- Multiple columns
SELECT age, gender, COUNT(*) 
FROM students 
GROUP BY age, gender;

üîí Constraints

PRIMARY KEY
‚úì Fully Supported in MySQL 
CREATE TABLE students (
    id INT PRIMARY KEY AUTO_INCREMENT,
    -- or
    id INT AUTO_INCREMENT,
    name VARCHAR(100),
    PRIMARY KEY (id)
    -- or composite key
    -- PRIMARY KEY (id, name)
);
FOREIGN KEY
‚úì Fully Supported in MySQL 
CREATE TABLE enrollments (
    id INT PRIMARY KEY AUTO_INCREMENT,
    student_id INT,
    course_id INT,
    FOREIGN KEY (student_id) REFERENCES students(id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    FOREIGN KEY (course_id) REFERENCES courses(id)
);
UNIQUE
‚úì Fully Supported in MySQL 
CREATE TABLE students (
    id INT PRIMARY KEY,
    email VARCHAR(100) UNIQUE,
    phone VARCHAR(15),
    UNIQUE (phone)
);
NOT NULL
‚úì Fully Supported in MySQL 
CREATE TABLE students (
    id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    age INT NOT NULL
);
CHECK
‚úì Supported in MySQL 8.0.16+
Note: CHECK constraints are only enforced in MySQL 8.0.16 and later. In earlier versions, they are parsed but ignored. 
CREATE TABLE students (
    id INT PRIMARY KEY,
    age INT CHECK (age >= 18),
    grade CHAR(1) CHECK (grade IN ('A', 'B', 'C', 'D', 'F'))
);
Alternative for older MySQL: Use triggers or application-level validation
DEFAULT
‚úì Fully Supported in MySQL 
CREATE TABLE students (
    id INT PRIMARY KEY,
    name VARCHAR(100),
    status VARCHAR(20) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP

üîó JOIN Operations

‚úì Fully Supported in MySQL

INNER JOIN courses ON enrollments.course_id = courses.id;
LEFT JOIN (LEFT OUTER JOIN)
‚úì Fully Supported in MySQL

FROM students
LEFT JOIN enrollments ON students.id = enrollments.student_id
LEFT JOIN courses ON enrollments.course_id = courses.id;
RIGHT JOIN (RIGHT OUTER JOIN)
‚úì Fully Supported in MySQL

FROM students
RIGHT JOIN enrollments ON students.id = enrollments.student_id;
CROSS JOIN
‚úì Fully Supported in MySQL 
SELECT students.name, courses.title
FROM students
CROSS JOIN courses;
SELF JOIN
‚úì Fully Supported in MySQL 
SELECT e1.name AS employee, e2.name AS manager
FROM employees e1
LEFT JOIN employees e2 ON e1.manager_id = e2.id;
FULL OUTER JOIN
‚úó Not Supported in MySQL
MySQL Alternative: Use UNION of LEFT and RIGHT JOIN 
SELECT students.name, courses.title
FROM students
LEFT JOIN enrollments ON students.id = enrollments.student_id
UNION
SELECT students.name, courses.title
FROM students
RIGHT JOIN enrollments ON students.id = enrollments.student_id;

Window Functions (Analytical Functions)
‚úì Supported in MySQL 8.0+
Note: Window functions are only available in MySQL 8.0 and later versions. 
-- ROW_NUMBER
SELECT name, age, 
       ROW_NUMBER() OVER (ORDER BY age DESC) AS row_num
FROM students;

-- RANK / DENSE_RANK
SELECT name, score,
       RANK() OVER (ORDER BY score DESC) AS rank,
       DENSE_RANK() OVER (ORDER BY score DESC) AS dense_rank
FROM students;

-- PARTITION BY
SELECT department, name, salary,
       ROW_NUMBER() OVER (PARTITION BY department ORDER BY salary DESC) AS dept_rank
FROM employees;

-- LAG / LEAD
SELECT name, salary,
       LAG(salary) OVER (ORDER BY id) AS prev_salary,
       LEAD(salary) OVER (ORDER BY id) AS next_salary
FROM employees;

-- NTILE (divide into groups)
SELECT name, score,
       NTILE(4) OVER (ORDER BY score DESC) AS quartile
FROM students;
Alternative for MySQL 5.x: Use variables for row numbering 
SET @row_num = 0;
SELECT name, age, (@row_num := @row_num + 1) AS row_num
FROM students
ORDER BY age DESC;
Common Table Expressions (CTE)
‚úì Supported in MySQL 8.0+
Note: CTEs are only available in MySQL 8.0 and later versions. 
-- Basic CTE
WITH student_avg AS (
    SELECT department, AVG(score) AS avg_score
    FROM students
    GROUP BY department
)
SELECT * FROM student_avg WHERE avg_score > 80;

-- Multiple CTEs
WITH 
    high_scorers AS (SELECT * FROM students WHERE score > 90),
    low_scorers AS (SELECT * FROM students WHERE score < 50)
SELECT 'High' AS category, COUNT(*) FROM high_scorers
UNION ALL
SELECT 'Low' AS category, COUNT(*) FROM low_scorers;

-- Recursive CTE
WITH RECURSIVE numbers AS (
    SELECT 1 AS n
    UNION ALL
    SELECT n + 1 FROM numbers WHERE n < 10
)
SELECT * FROM numbers;
Alternative for MySQL 5.x: Use subqueries or temporary tables 
-- Using subquery
SELECT * FROM (
    SELECT department, AVG(score) AS avg_score
    FROM students
    GROUP BY department
) AS student_avg
WHERE avg_score > 80;

-- Using temporary table
CREATE TEMPORARY TABLE student_avg AS
SELECT department, AVG(score) AS avg_score
FROM students
GROUP BY department;

SELECT * FROM student_avg WHERE avg_score > 80;
DROP TEMPORARY TABLE student_avg;

Performance & Optimization Tips
‚úì Best Practices for MySQL
Key Performance Tips:
  • Always use indexes on columns in WHERE, JOIN, and ORDER BY clauses
  • Use EXPLAIN to analyze query performance
  • Avoid SELECT * - specify only needed columns
  • Use LIMIT for large result sets
  • Use appropriate data types (TINYINT instead of INT for small numbers)
  • Use JOIN instead of subqueries when possible
  • Enable query cache for frequently accessed data
  • Use prepared statements to prevent SQL injection
  • Regular database maintenance (OPTIMIZE TABLE, ANALYZE TABLE)
-- Optimize table
OPTIMIZE TABLE students;

-- Analyze table
ANALYZE TABLE students;

-- Check table
CHECK TABLE students;

-- Repair table
REPAIR TABLE students;

-- Show table size
SELECT 
    table_name,
    ROUND(((data_length + index_length) / 1024 / 1024), 2) AS "Size (MB)"
FROM information_schema.TABLES
WHERE table_schema = 'school'
ORDER BY (data_length + index_length) DESC;

Full-Text Search
‚úì Fully Supported in MySQL 
-- Create fulltext index
CREATE FULLTEXT INDEX idx_content ON articles(title, body);

-- Or add to existing table
ALTER TABLE articles ADD FULLTEXT(title, body);

-- Natural language search
SELECT * FROM articles
WHERE MATCH(title, body) AGAINST('mysql database');

-- Boolean search
SELECT * FROM articles
WHERE MATCH(title, body) AGAINST('+mysql -oracle' IN BOOLEAN MODE);

-- Query expansion
SELECT * FROM articles
WHERE MATCH(title, body) AGAINST('database' WITH QUERY EXPANSION);
